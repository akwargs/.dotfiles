	clean = git-lfs clean -- %f
	smudge = git-lfs smudge -- %f
	process = git-lfs filter-process
	required = true

	hist = log --all --decorate --oneline --graph
	lg = log --pretty='%C(red)%h%Creset %C(black)%d %s%Creset %C(blue)(%ar)%Creset'
	new = lg main..HEAD
	missing = lg HEAD..main

	rebase = true

	autocrlf = false
	editor = vim
